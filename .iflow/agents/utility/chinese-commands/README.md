## 🔧 指令自定义

### 创建自定义指令
```python
def create_custom_command(command_name, command_action, description=""):
    """
    创建自定义指令
    
    Args:
        command_name: 指令名称
        command_action: 指令动作
        description: 指令描述
    """
    
    custom_command = {
        'name': command_name,
        'action': command_action,
        'description': description,
        'created_at': get_current_timestamp(),
        'usage_count': 0
    }
    
    # 保存到指令库
    save_custom_command(custom_command)
    
    return f"自定义指令 '{command_name}' 创建成功！"
```

### 指令别名管理
```python
def create_command_alias(original_command, alias_name):
    """
    创建指令别名
    
    Args:
        original_command: 原始指令
        alias_name: 别名名称
    """
    
    alias_mapping = {
        'original': original_command,
        'alias': alias_name,
        'created_at': get_current_timestamp()
    }
    
    # 保存别名映射
    save_command_alias(alias_mapping)
    
    return f"指令别名 '{alias_name}' -> '{original_command}' 创建成功！"
```
